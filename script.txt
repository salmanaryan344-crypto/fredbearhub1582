local currentTheme = "Black"
local currentTab = "main"

-- Update Tab Colors
local function updateTabColors()
    local theme = themes[currentTheme]
    local buttons = {mainTabBtn, settingsTabBtn, creditsTabBtn, logsTabBtn}
    
    for i, btn in ipairs(buttons) do
        local isActive = (i == 1 and currentTab == "main") or 
                        (i == 2 and currentTab == "settings") or 
                        (i == 3 and currentTab == "credits") or 
                        (i == 4 and currentTab == "logs")
        
        TweenService:Create(btn, TweenInfo.new(0.3), {
            BackgroundColor3 = isActive and Color3.fromRGB(
                math.floor(theme.buttonColor.R * 255 * 1.2),
                math.floor(theme.buttonColor.G * 255 * 1.2),
                math.floor(theme.buttonColor.B * 255 * 1.2)
            ) or Color3.fromRGB(
                math.floor(theme.buttonColor.R * 255 * 0.8),
                math.floor(theme.buttonColor.G * 255 * 0.8),
                math.floor(theme.buttonColor.B * 255 * 0.8)
            ),
            TextColor3 = isActive and Color3.new(1, 1, 1) or Color3.new(0.7, 0.7, 0.7)
        }):Play()
    end
end

-- Tab Switching
local function switchTab(tabName)
    currentTab = tabName
    
    mainContent.Visible = false
    settingsContent.Visible = false
    creditsContent.Visible = false
    logsContent.Visible = false
    
    if tabName == "main" then
        mainContent.Visible = true
    elseif tabName == "settings" then
        settingsContent.Visible = true
    elseif tabName == "credits" then
        creditsContent.Visible = true
    elseif tabName == "logs" then
        logsContent.Visible = true
    end
    
    updateTabColors()
end

mainTabBtn.MouseButton1Click:Connect(function() switchTab("main") end)

local highlightEnabled = false
local function toggleHighlight()
	highlightEnabled = not highlightEnabled
	highlightBtn.Text = "Highlight Animatronics: " .. (highlightEnabled and "ON" or "OFF")
	
	for _, name in ipairs({"FredBear", "SpringBonnie"}) do
		local model = workspace:FindFirstChild(name)
		if model then
			local h = model:FindFirstChild("AnimHighlight")
			if highlightEnabled and not h then
				h = Instance.new("Highlight")
				h.Name = "AnimHighlight"
				h.FillTransparency = 0.8
				h.FillColor = Color3.new(1, 1, 1)
				h.OutlineColor = Color3.new(1, 1, 1)
				h.OutlineTransparency = 0
				h.Adornee = model
				h.Parent = model
			elseif not highlightEnabled and h then
				h:Destroy()
			end
		end
	end
end

local MIN_SPEED, MAX_SPEED = 16, 100
local draggingSlider, currentSpeed = false, 16

local function updateSpeedFromPos(x)
	x = math.clamp(x, 0, sliderFrame.AbsoluteSize.X)
	local speed = math.floor(MIN_SPEED + (x / sliderFrame.AbsoluteSize.X) * (MAX_SPEED - MIN_SPEED))
	currentSpeed = speed
	fill.Size = UDim2.new(0,x,1,0)
	handle.Position = UDim2.new(0,x,0.5,0)
	local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
	if hum then hum.WalkSpeed = speed end
	speedLabel.Text = "Speed Adjuster: "..speed
end

sliderFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		draggingSlider = true
		updateSpeedFromPos(input.Position.X - sliderFrame.AbsolutePosition.X)
	end
end)
sliderFrame.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		draggingSlider = false
	end
end)
UserInputService.InputChanged:Connect(function(input)
	if draggingSlider and input.UserInputType == Enum.UserInputType.MouseMovement then
		updateSpeedFromPos(input.Position.X - sliderFrame.AbsolutePosition.X)
	end
end)

local function applySpeedOnRespawn()
	local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
	if hum then
		hum.WalkSpeed = currentSpeed
		local x = ((currentSpeed - MIN_SPEED) / (MAX_SPEED - MIN_SPEED)) * sliderFrame.AbsoluteSize.X
		fill.Size = UDim2.new(0,x,1,0)
		handle.Position = UDim2.new(0,x,0.5,0)
		speedLabel.Text = "Speed Adjuster: "..currentSpeed
	end
end
player.CharacterAdded:Connect(function() task.wait(0.2); applySpeedOnRespawn() end)

local fullBrightEnabled = false
local function toggleFullBright()
	fullBrightEnabled = not fullBrightEnabled
	if fullBrightEnabled then
		Lighting.Brightness = 3
		Lighting.Ambient = Color3.new(1, 1, 1)
		Lighting.OutdoorAmbient = Color3.new(1, 1, 1)
		local bloom = Lighting:FindFirstChild("FullBrightBloom") or Instance.new("BloomEffect")
		bloom.Name = "FullBrightBloom"
		bloom.Intensity = 0.5
		bloom.Threshold = 0.8
		bloom.Size = 24
		bloom.Parent = Lighting
		fullBrightBtn.Text = "Full Bright: ON"
	else
		Lighting.Brightness = 1
		Lighting.Ambient = Color3.fromRGB(128, 128, 128)
		Lighting.OutdoorAmbient = Color3.fromRGB(128, 128, 128)
		local bloom = Lighting:FindFirstChild("FullBrightBloom")
		if bloom then bloom:Destroy() end
		fullBrightBtn.Text = "Full Bright: OFF"
	end
end

-- Theme Buttons
local themeButtons = {}
local function createThemeButton(themeName, yPos)
	local btn = Instance.new("TextButton")
	btn.Size = UDim2.new(0.9, 0, 0, 40)
	btn.Position = UDim2.new(0.05, 0, 0, yPos)
	btn.BackgroundColor3 = themes[themeName].buttonColor
	btn.Text = themeName
	btn.TextColor3 = Color3.new(1, 1, 1)
	btn.Font = Enum.Font.Gotham
	btn.TextSize = 15
	btn.Parent = settingsContainer
	Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 12)
	
	if themeName == currentTheme then
		btn.BackgroundColor3 = themes[themeName].accentColor
	end
	
	btn.MouseButton1Click:Connect(function()
		currentTheme = themeName
		local theme = themes[themeName]
		
		TweenService:Create(frame, TweenInfo.new(0.3), {BackgroundColor3 = theme.frameColor}):Play()
		TweenService:Create(sliderFrame, TweenInfo.new(0.3), {BackgroundColor3 = theme.sliderColor}):Play()
		TweenService:Create(fill, TweenInfo.new(0.3), {BackgroundColor3 = theme.accentColor}):Play()
		
		for _, button in ipairs({highlightBtn, resetBtn, fullBrightBtn, noFogBtn}) do
			if button then
				TweenService:Create(button, TweenInfo.new(0.3), {BackgroundColor3 = theme.buttonColor}):Play()
			end
		end
		
		currentThemeLabel.Text = "Current Theme: " .. themeName
		updateTabColors()
		
		for name, tbtn in pairs(themeButtons) do
			if name == themeName then
				tbtn.BackgroundColor3 = theme.accentColor
			else
				tbtn.BackgroundColor3 = themes[name].buttonColor
			end
		end
	end)
	
	themeButtons[themeName] = btn
	return btn
end

-- Draggable GUI
local dragging, dragStart, startPos
frame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		local mousePos = input.Position
		if mousePos.Y < frame.AbsolutePosition.Y + 45 then
			dragging = true
			dragStart = mousePos
			startPos = frame.Position
		end
	end
end)
frame.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = false
	end
end)
UserInputService.InputChanged:Connect(function(input)
	if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
	end
end)

-- Minimize
local minimized = false
local originalSize = frame.Size
minimize.MouseButton1Click:Connect(function()
	minimized = not minimized
	if minimized then
		TweenService:Create(frame, TweenInfo.new(0.3), {Size = UDim2.new(frame.Size.X.Scale, frame.Size.X.Offset, 0, 95)}):Play()
		tabContent.Visible = false
	else
		TweenService:Create(frame, TweenInfo.new(0.3), {Size = originalSize}):Play()
		tabContent.Visible = true
		switchTab(currentTab)
	end
end)
